465  , typename CT > // Type of the condition variable
466 ThreadPool<TT,MT,LT,CT>::~ThreadPool()
467 {
468  Lock lock( mutex_ );
469 
470  // Removing all currently queued tasks
471  taskqueue_.clear();
472 
473  // Setting the expected number of threads
474  expected_ = 0;
475 
476  // Notifying all idle threads
477  waitForTask_.notify_all();
478 
479  // Waiting for all threads to terminate
480  while( total_ != 0 ) {
481  waitForThread_.wait( lock );
482  }
483 
484  // Joining all threads
485  for( typename Threads::Iterator thread=threads_.begin(); thread!=threads_.end(); ++thread ) {
486  thread->join();
487  }
488 
489  // Destroying all threads
490  threads_.clear();
491 }

799 void ThreadPool<TT,MT,LT,CT>::resize( size_t n, bool block )
800 {
801  // Checking the given number of threads
802 #if !(defined _MSC_VER)
803  if( n == 0 )
804  throw std::invalid_argument( "Invalid number of threads" );
805 #endif
806 
807  // Adjusting the number of threads
808  {
809  Lock lock( mutex_ );
810 
811  // Adding new threads to the thread pool
812  if( n > expected_ ) {
813  for( size_t i=expected_; i<n; ++i )
814  createThread();
815  }
816 
817  // Removing threads from the pool
818  else {
819  expected_ = n;
820  waitForTask_.notify_all();
821 
822  while( block && total_ != expected_ ) {
823  waitForThread_.wait( lock );
824  }
825  }
826 
827  // Joining and destroying any terminated thread
828  for( typename Threads::Iterator thread=threads_.begin(); thread!=threads_.end(); ) {
829  if( thread->hasTerminated() ) {
830  thread->join();
831  thread = threads_.erase( thread );
832  }
833  else ++thread;
834  }
835  }
836 }
